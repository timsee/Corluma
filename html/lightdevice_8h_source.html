130 QColor static colorTemperatureToRGB(int ct) {
131  // convert to kelvin
132  float kelvin = (int)(1.0f / ct * 1000000.0f);
133  float temp = kelvin / 100;
134 
135  float red, green, blue;
136  if (temp <= 66) {
137  red = 255;
138  green = temp;
139  green = 99.4708025861 * std::log(green) - 161.1195681661;
140  if (temp <= 19) {
141  blue = 0;
142  } else {
143  blue = temp - 10;
144  blue = 138.5177312231 * std::log(blue) - 305.0447927307;
145  }
146  } else {
147  red = temp - 60;
148  red = 329.698727446 * std::pow(red, -0.1332047592);
149 
150  green = temp - 60;
151  green = 288.1221695283 * std::pow(green, -0.0755148492);
152 
153  blue = 255;
154  }
155 
156  red = std::max(0.0f, std::min(red, 255.0f));
157  green = std::max(0.0f, std::min(green, 255.0f));
158  blue = std::max(0.0f, std::min(blue, 255.0f));
159  return QColor(red, green, blue);
160 }

168 int static rgbToColorTemperature(QColor color) {
169  float minTemperature = 153;
170  float maxTemperature = 500;
171  for (int i = minTemperature; i <= maxTemperature; ++i) {
172  QColor testColor = colorTemperatureToRGB(i);
173  float r = std::abs(testColor.red() - color.red()) / 255.0f;
174  float g = std::abs(testColor.green() - color.green()) / 255.0f;
175  float b = std::abs(testColor.blue() - color.blue()) / 255.0f;
176  float difference = (r + g + b) / 3.0f;
177  if (difference < 0.02f) {
178  return i;
179  }
180  }
181  return -1;
182 }

291  SLightDevice static stringToStruct(QString string) {
292  // first split the values from comma delimited to a vector of strings
293  std::vector<std::string> valueVector;
294  std::stringstream input(string.toStdString());
295  while (input.good()) {
296  std::string value;
297  std::getline(input, value, ',');
298  valueVector.push_back(value);
299  }
300  // check validity
301  SLightDevice outputStruct;
302  if (valueVector.size() == 3) {
303  outputStruct.name = QString::fromStdString(valueVector[0]);
304  outputStruct.index = QString::fromStdString(valueVector[1]).toInt();
305  outputStruct.type = (ECommType)QString::fromStdString(valueVector[2]).toInt();
306  if (outputStruct.type == ECommType::eHue) {
307  outputStruct.name = "Bridge";
308  }
309  } else {
310  qDebug() << "something went wrong with the key...";
311  }
312  return outputStruct;
313  }
