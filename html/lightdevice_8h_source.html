144  SLightDevice static stringToStruct(QString string) {
145  // first split the values from comma delimited to a vector of strings
146  std::vector<std::string> valueVector;
147  std::stringstream input(string.toStdString());
148  while (input.good()) {
149  std::string value;
150  std::getline(input, value, ',');
151  valueVector.push_back(value);
152  }
153  // check validity
154  SLightDevice outputStruct;
155  if (valueVector.size() == 3) {
156  outputStruct.name = QString::fromStdString(valueVector[0]);
157  outputStruct.index = QString::fromStdString(valueVector[1]).toInt();
158  outputStruct.type = (ECommType)QString::fromStdString(valueVector[2]).toInt();
159  if (outputStruct.type == ECommType::eHue) {
160  outputStruct.name = "Bridge";
161  }
162  } else {
163  qDebug() << "something went wrong with the key...";
164  }
165  return outputStruct;
166  }
