159 SLightDevice static stringToStruct(QString string) {
160  // first split the values from comma delimited to a vector of strings
161  std::vector<std::string> valueVector;
162  std::stringstream input(string.toStdString());
163  while (input.good()) {
164  std::string value;
165  std::getline(input, value, ',');
166  valueVector.push_back(value);
167  }
168  // check validity
169  SLightDevice outputStruct;
170  if (valueVector.size() == 3) {
171  outputStruct.name = QString::fromStdString(valueVector[0]);
172  outputStruct.index = QString::fromStdString(valueVector[1]).toInt();
173  outputStruct.type = (ECommType)QString::fromStdString(valueVector[2]).toInt();
174  if (outputStruct.type == ECommType::eHue) {
175  outputStruct.name = "Bridge";
176  }
177  } else {
178  qDebug() << "something went wrong with the key...";
179  }
180  return outputStruct;
181 }
