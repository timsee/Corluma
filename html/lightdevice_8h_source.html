171 inline QColor colorTemperatureToRGB(int ct) {
172  // convert to kelvin
173  float kelvin = (int)(1.0f / ct * 1000000.0f);
174  float temp = kelvin / 100;
175 
176  float red, green, blue;
177  if (temp <= 66) {
178  red = 255;
179  green = temp;
180  green = 99.4708025861 * std::log(green) - 161.1195681661;
181  if (temp <= 19) {
182  blue = 0;
183  } else {
184  blue = temp - 10;
185  blue = 138.5177312231 * std::log(blue) - 305.0447927307;
186  }
187  } else {
188  red = temp - 60;
189  red = 329.698727446 * std::pow(red, -0.1332047592);
190 
191  green = temp - 60;
192  green = 288.1221695283 * std::pow(green, -0.0755148492);
193 
194  blue = 255;
195  }
196 
197  red = std::max(0.0f, std::min(red, 255.0f));
198  green = std::max(0.0f, std::min(green, 255.0f));
199  blue = std::max(0.0f, std::min(blue, 255.0f));
200  return QColor(red, green, blue);
201 }

209 inline int rgbToColorTemperature(QColor color) {
210  float minTemperature = 153;
211  float maxTemperature = 500;
212  for (int i = minTemperature; i <= maxTemperature; ++i) {
213  QColor testColor = colorTemperatureToRGB(i);
214  float r = std::abs(testColor.red() - color.red()) / 255.0f;
215  float g = std::abs(testColor.green() - color.green()) / 255.0f;
216  float b = std::abs(testColor.blue() - color.blue()) / 255.0f;
217  float difference = (r + g + b) / 3.0f;
218  if (difference < 0.02f) {
219  return i;
220  }
221  }
222  return -1;
223 }
