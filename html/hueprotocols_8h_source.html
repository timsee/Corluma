211 inline EHueType checkForHueWithMostFeatures(std::list<SHueLight> lights) {
212  uint32_t ambientCount = 0;
213  uint32_t whiteCount = 0;
214  uint32_t rgbCount = 0;
215  // check for all devices
216  for (auto&& hue : lights) {
217  // check if its a hue
218  if (hue.type == EHueType::eExtended
219  || hue.type == EHueType::eColor) {
220  rgbCount++;
221  } else if (hue.type == EHueType::eAmbient) {
222  ambientCount++;
223  } else if (hue.type == EHueType::eWhite) {
224  whiteCount++;
225  }
226  }
227 
228  if (whiteCount > 0
229  && (ambientCount == 0)
230  && (rgbCount == 0)) {
231  return EHueType::eWhite;
232  }
233  if (ambientCount > 0
234  && (rgbCount == 0)) {
235  return EHueType::eAmbient;
236  }
237 
238  if (rgbCount > 0) {
239  return EHueType::eExtended;
240  }
241 
242  return EHueType::EHueType_MAX;
243 }
