213 inline EHueType checkForHueWithMostFeatures(std::list<SHueLight> lights) {
214  uint32_t ambientCount = 0;
215  uint32_t whiteCount = 0;
216  uint32_t rgbCount = 0;
217  // check for all devices
218  for (auto&& hue : lights) {
219  // check if its a hue
220  if (hue.type == EHueType::eExtended
221  || hue.type == EHueType::eColor) {
222  rgbCount++;
223  } else if (hue.type == EHueType::eAmbient) {
224  ambientCount++;
225  } else if (hue.type == EHueType::eWhite) {
226  whiteCount++;
227  }
228  }
229 
230  if (whiteCount > 0
231  && (ambientCount == 0)
232  && (rgbCount == 0)) {
233  return EHueType::eWhite;
234  }
235  if (ambientCount > 0
236  && (rgbCount == 0)) {
237  return EHueType::eAmbient;
238  }
239 
240  if (rgbCount > 0) {
241  return EHueType::eExtended;
242  }
243 
244  return EHueType::EHueType_MAX;
245 }
