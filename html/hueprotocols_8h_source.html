195 inline EHueType checkForHueWithMostFeatures(std::list<SHueLight> lights) {
196  uint32_t ambientCount = 0;
197  uint32_t whiteCount = 0;
198  uint32_t rgbCount = 0;
199  // check for all devices
200  for (auto&& hue : lights) {
201  // check if its a hue
202  if (hue.type == EHueType::eExtended
203  || hue.type == EHueType::eColor) {
204  rgbCount++;
205  } else if (hue.type == EHueType::eAmbient) {
206  ambientCount++;
207  } else if (hue.type == EHueType::eWhite) {
208  whiteCount++;
209  }
210  }
211 
212  if (whiteCount > 0
213  && (ambientCount == 0)
214  && (rgbCount == 0)) {
215  return EHueType::eWhite;
216  }
217  if (ambientCount > 0
218  && (rgbCount == 0)) {
219  return EHueType::eAmbient;
220  }
221 
222  if (rgbCount > 0) {
223  return EHueType::eExtended;
224  }
225 
226  return EHueType::EHueType_MAX;
227 }
