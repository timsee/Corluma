60  void updateNanoleaf(const nano::LeafMetadata& metadata, nano::ELeafDiscoveryState status) {
61  mLeafMetadata = metadata;
62  // pick up when status changes for edge cases.
63  bool statusChanged = (status != mStatus);
64  mStatus = status;
65  handleStatusPrompt();
66  if (statusChanged) {
67  // resize when status changes
68  resize();
69  }
70  if (mStatus == nano::ELeafDiscoveryState::searchingIP
71  || mStatus == nano::ELeafDiscoveryState::searchingAuth
72  || mStatus == nano::ELeafDiscoveryState::reverifying) {
73  // show a syncing widget, we haven't found it yet
74  mSyncWidget->setVisible(true);
75  mStatusPrompt->setVisible(true);
76  mSyncWidget->changeState(ESyncState::syncing);
77  } else {
78  mSyncWidget->setVisible(false);
79  mStatusPrompt->setVisible(true);
80 
81  mStatus = status;
82  mLeafMetadata = metadata;
83  if (!mLeafMetadata.name().isEmpty()) {
84  mName->setText(mLeafMetadata.name());
85  } else {

88  mName->setText(mLeafMetadata.hardwareName());
89  } else {
90  mName->setText(mLeafMetadata.IP());

94  drawPanels();
