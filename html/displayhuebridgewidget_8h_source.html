274  auto lights = mComm->lightsByIDs(uniqueIDs);
275  for (auto light : lights) {
276  if (light.isValid()) {
277  if (light.protocol() == EProtocolType::hue) {
278  auto metadata = mComm->hue()->metadataFromLight(light);
279  mLightInfoWidget->addLight(metadata);
280  mLightInfoWidget->changeRowHeight(mRowHeight);
281  mLights->addLight(light);
282  // update the bridge of this hue
283  auto bridge = mComm->hue()->discovery()->bridgeFromLight(metadata);
284  updateBridge(bridge);
285  handleState();
286  }
287  } else {
288  qDebug() << " INFO: invalid light found from discovery... this shouldn't happen..."
289  << light;
290  }
291  }
