191  static std::vector<Group> mergeLightGroups(const std::vector<Group>& first,
192  const std::vector<Group>& second) {

194  for (const auto& group : second) {
195  // look for groups with same name and type
196  auto result = std::find(retList.begin(), retList.end(), group);
197  if (result == retList.end()) {
198  // if one doesn't exist, add it
199  retList.push_back(group);
200  } else {
201  // if one does exist, merge the devices instead
202  Group combinedGroup = *result;
203  auto lights = combinedGroup.lights();
204  lights.insert(lights.end(), group.lights().begin(), group.lights().end());
205  combinedGroup.lights(lights);
206  retList.erase(result);

208  }
