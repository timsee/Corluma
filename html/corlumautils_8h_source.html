66  green = 99.4708025861 * std::log(green) - 161.1195681661;

95 inline int rgbToColorTemperature(QColor color) {
96  float minTemperature = 153;
97  float maxTemperature = 500;
98  for (int i = minTemperature; i <= maxTemperature; ++i) {
99  QColor testColor = colorTemperatureToRGB(i);
100  float r = std::abs(testColor.red() - color.red()) / 255.0f;
101  float g = std::abs(testColor.green() - color.green()) / 255.0f;
102  float b = std::abs(testColor.blue() - color.blue()) / 255.0f;
103  float difference = (r + g + b) / 3.0f;
104  if (difference < 0.02f) {
105  return i;
106  }
107  }
108  return -1;
