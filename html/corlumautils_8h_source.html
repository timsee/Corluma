46 inline QColor colorTemperatureToRGB(int ct) {
47  // convert to kelvin
48  float kelvin = (int)(1.0f / ct * 1000000.0f);
49  float temp = kelvin / 100;
50 
51  float red, green, blue;
52  if (temp <= 66) {
53  red = 255;
54  green = temp;
55  green = 99.4708025861 * std::log(green) - 161.1195681661;
56  if (temp <= 19) {
57  blue = 0;
58  } else {
59  blue = temp - 10;
60  blue = 138.5177312231 * std::log(blue) - 305.0447927307;
61  }
62  } else {
63  red = temp - 60;
64  red = 329.698727446 * std::pow(red, -0.1332047592);
65 
66  green = temp - 60;
67  green = 288.1221695283 * std::pow(green, -0.0755148492);
68 
69  blue = 255;
70  }
71 
72  red = std::max(0.0f, std::min(red, 255.0f));
73  green = std::max(0.0f, std::min(green, 255.0f));
74  blue = std::max(0.0f, std::min(blue, 255.0f));

84 inline int rgbToColorTemperature(QColor color) {
85  float minTemperature = 153;
86  float maxTemperature = 500;
87  for (int i = minTemperature; i <= maxTemperature; ++i) {
88  QColor testColor = colorTemperatureToRGB(i);
89  float r = std::abs(testColor.red() - color.red()) / 255.0f;
90  float g = std::abs(testColor.green() - color.green()) / 255.0f;
91  float b = std::abs(testColor.blue() - color.blue()) / 255.0f;
92  float difference = (r + g + b) / 3.0f;
93  if (difference < 0.02f) {
94  return i;
95  }
96  }
97  return -1;
98 }
