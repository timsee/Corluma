188 inline EHueType checkForHueWithMostFeatures(std::vector<HueMetadata> lights) {
189  std::uint32_t ambientCount = 0;
190  std::uint32_t whiteCount = 0;
191  std::uint32_t rgbCount = 0;
192  // check for all devices
193  for (const auto& hue : lights) {
194  // check if its a hue
195  if (hue.hueType() == EHueType::extended || hue.hueType() == EHueType::color) {
196  rgbCount++;
197  } else if (hue.hueType() == EHueType::ambient) {
198  ambientCount++;
199  } else if (hue.hueType() == EHueType::white) {
200  whiteCount++;
201  }
202  }
203 
204  if (whiteCount > 0 && (ambientCount == 0) && (rgbCount == 0)) {
205  return EHueType::white;
206  }
207  if (ambientCount > 0 && (rgbCount == 0)) {
208  return EHueType::ambient;
209  }
210 
211  if (rgbCount > 0) {
212  return EHueType::extended;
213  }
214 
215  return EHueType::MAX;
216 }
