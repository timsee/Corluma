83  bool insert(const std::string& key, const T& item) {
84  // search to see if key already exists
85  auto keyIterator = mKeyToItemMap.find(key);
86  bool keyExists = (keyIterator != mKeyToItemMap.end());
87  // search to see if item already exists
88  auto itemIterator = mItemToKeyMap.find(item);
89  bool itemExists = (itemIterator != mItemToKeyMap.end());
90  // if either exists already, return false
91  if (keyExists || itemExists) {
92  return false;
93  } else {
94  // if neither exists, add to both.
95  auto insertIterator = mKeyToItemMap.emplace(key, item);
96  mItemToKeyMap.emplace(item, insertIterator.first->first);
97  GUARD_EXCEPTION(mItemToKeyMap.size() == mKeyToItemMap.size(),
98  "key map and item map don't match in size");
99  return true;
100  }
101  }
