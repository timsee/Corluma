105  void addGroupsToLeftMenu() {
106  auto groups = mGroups->groupDict().items();
107  auto knownStateArgs = mGroupStates->groupStates();
108  mGroupsWidget->clear();
109  for (const auto& group : groups) {
110  bool foundStateArgs = false;
111  for (const auto& knownStateArg : knownStateArgs) {
112  if (knownStateArg.uniqueID() == group.uniqueID()) {
113  foundStateArgs = true;
114  }
115  }
116  if (!foundStateArgs) {
117  // ignore groups that are already in the default states
118  cor::GroupState groupState(group.uniqueID(), {});
119  groupState.name(group.name());
120  mGroupsWidget->addState(groupState);
121  }
122  }
