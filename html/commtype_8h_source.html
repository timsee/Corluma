295  SLightDevice static stringToStruct(QString string) {
296  // first split the values from comma delimited to a vector of strings
297  std::vector<std::string> valueVector;
298  std::stringstream input(string.toStdString());
299  while (input.good()) {
300  std::string value;
301  std::getline(input, value, ',');
302  valueVector.push_back(value);
303  }
304  // check validity
305  SLightDevice outputStruct;
306  if (valueVector.size() == 3) {
307  outputStruct.name = QString::fromStdString(valueVector[0]);
308  outputStruct.index = QString::fromStdString(valueVector[1]).toInt();
309  outputStruct.type = (ECommType)QString::fromStdString(valueVector[2]).toInt();
310  if (outputStruct.type == ECommType::eHue) {
311  outputStruct.name = "Bridge";
312  }
313  } else {
314  qDebug() << "something went wrong with the key...";
315  }
316  return outputStruct;
317  }
