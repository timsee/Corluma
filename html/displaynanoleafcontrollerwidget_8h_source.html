351  void updateMetadata(const nano::LeafMetadata& leafMetadata) {
352  std::stringstream returnString;
353 
354  if (mDiscoveryState == nano::ELeafDiscoveryState::connected) {
355  returnString << "<b>Model:</b> " << leafMetadata.model().toStdString() << "<br>";
356  returnString << "<b>Firmware:</b> " << leafMetadata.firmware().toStdString() << "<br>";
357  returnString << "<b>Serial:</b> " << leafMetadata.serialNumber().toStdString()
358  << "<br><br>";
359  }
360 
361  returnString << "<b>IP:</b> " << leafMetadata.IP().toStdString() << "<br>";
362  returnString << "<b>Port:</b> " << leafMetadata.port() << "<br><br>";
363  returnString << "<b>Hardware Name:</b> " << leafMetadata.hardwareName().toStdString()
364  << "<br>";
365  if (mDiscoveryState == nano::ELeafDiscoveryState::connected) {
366  returnString << "<b>Hardware Version:</b> "
367  << leafMetadata.hardwareVersion().toStdString() << "<br>";
368 
369  if (leafMetadata.rhythmController().isConnected()) {
370  auto rhythmController = leafMetadata.rhythmController();
371  returnString << "<br><br><b>Rhythm Controller</b><br>";
372  returnString << "<b>ID: </b>" << rhythmController.ID() << "<br>";
373  returnString << "<b>Firmware: </b>"
374  << rhythmController.firmwareVersion().toStdString() << "<br>";
375  returnString << "<b>Hardware Version: </b>"
376  << rhythmController.hardwareVersion().toStdString() << "<br>";
377  returnString << "<b>Mode: </b>" << rhythmController.mode() << "<br>";
378 
379  if (rhythmController.auxAvailable()) {
380  returnString << "<b>Aux Detected</b><br>";
381  } else {
382  returnString << "<b>No Aux Detected</b><br>";
383  }
384  } else if (leafMetadata.model() == "NL22") {
385  // NL22 requires a separate rhtym controller, in most newer lights, this feature is
386  // built in.
387  returnString << "<b>No Rhythm Controller Detected</b><br>";
388  }
389  }
390  mMetadata->updateText(QString(returnString.str().c_str()));
391  }
